      <p>ReAppearance begins with a simple claim: the <span class="neon">appearance</span> of architecture is not a residue of the past but a form of ongoing life. We work with what survives—fragments, scans, oral accounts, routine maintenance logs—and accept that visibility is unstable. Some things reappear by renewal, as at Ise where rebuilding is a pedagogy; some reappear by relocation, as with minka whose craft persists after displacement; some reappear only as documents, like the Sawin House whose image now exceeds its matter. The platform does not promise closure. It provides a public surface where these states of being can be read together without forcing them into one narrative of loss or one solution of restitution.</p>
      <p>Preservation here is a practice of composition. We arrange sources that were never meant to align: a floor plan beside a street photograph, a conservator’s note beside a neighbor’s memory, a timber joint beside a demolition permit. Each juxtaposition changes the weight of what came before and after. To publish these relations in the open, as lightweight browser scenes, is to argue that <span class="neon">accessibility</span> is a structural part of care. Materials are licensed to circulate; metadata is kept legible; absences are named. The work is not to reconstruct an origin but to maintain a field of relations where study can happen and where new accounts can enter without asking permission.</p>
      <p>ReAppearance treats replication as an ethic rather than an aesthetic effect. A copy that cites its sources can keep cultural techniques alive; a model that declares its gaps can invite counter-evidence; a record that can be forked can be stewarded by many. The platform is therefore minimal by design: portable files, clear credits, and public methods that others can adopt for their own sites. What we curate is not a definitive image of a building but a stable way to keep its appearances open—so that demolition does not equal erasure, ritual does not equal sameness, and relocation does not equal replacement. In this way, the archive behaves less like a vault and more like a <span class="neon">commons</span>.</p>

    // --- Simple bouncing animation for floatable elements ---
    (function() {
      const stage = document.getElementById('stage');
      const nodes = Array.from(stage.querySelectorAll('.floatable'));

      // Create state for each node
      const items = nodes.map((el, i) => {
        const rect = el.getBoundingClientRect();
        const stageRect = stage.getBoundingClientRect();
        // initial position from inline style or computed
        const style = window.getComputedStyle(el);
        let left = parseFloat(style.left) || (rect.left - stageRect.left);
        let top = parseFloat(style.top) || (rect.top - stageRect.top);
        const width = rect.width;
        const height = rect.height;
        // px per second speed, randomized within nice range
        const speed = 55 + Math.random() * 40; // 55–95 px/s (a bit faster, still readable) // 40–80 px/s (fast but not crazy)
        const angle = Math.random() * Math.PI * 2;
        let vx = Math.cos(angle) * speed;
        let vy = Math.sin(angle) * speed;
        return { el, left, top, width, height, vx, vy };
      });

      let last = performance.now();

      function step(now) {
        const dt = Math.min((now - last) / 1000, 0.05); // seconds, cap to keep stable
        last = now;
        const stageRect = stage.getBoundingClientRect();
        const W = stageRect.width;
        const H = stageRect.height;

        // Update positions
        for (let i = 0; i < items.length; i++) {
          const it = items[i];
          it.left += it.vx * dt;
          it.top += it.vy * dt;

          // Bounce on edges
          if (it.left <= 0) { it.left = 0; it.vx = Math.abs(it.vx); }
          if (it.top <= 0) { it.top = 0; it.vy = Math.abs(it.vy); }
          if (it.left + it.width >= W) { it.left = W - it.width; it.vx = -Math.abs(it.vx); }
          if (it.top + it.height >= H) { it.top = H - it.height; it.vy = -Math.abs(it.vy); }
        }

        // Naive collision detection (AABB). If overlapping, push apart + swap velocities
        for (let i = 0; i < items.length; i++) {
          for (let j = i + 1; j < items.length; j++) {
            const a = items[i], b = items[j];
            if (a.left < b.left + b.width && a.left + a.width > b.left && a.top < b.top + b.height && a.top + a.height > b.top) {
              // simple response: swap velocities and separate along the smallest axis
              const overlapX = (a.left + a.width) - b.left < (b.left + b.width) - a.left ? (a.left + a.width) - b.left : (b.left + b.width) - a.left;
              const overlapY = (a.top + a.height) - b.top < (b.top + b.height) - a.top ? (a.top + a.height) - b.top : (b.top + b.height) - a.top;
              if (overlapX < overlapY) {
                const shift = overlapX / 2;
                a.left -= shift; b.left += shift;
                const tmp = a.vx; a.vx = b.vx; b.vx = tmp;
              } else {
                const shift = overlapY / 2;
                a.top -= shift; b.top += shift;
                const tmp = a.vy; a.vy = b.vy; b.vy = tmp;
              }
            }
          }
        }

        // Apply transforms
        for (const it of items) {
          it.el.style.left = it.left + 'px';
          it.el.style.top = it.top + 'px';
        }

        requestAnimationFrame(step);
      }

      requestAnimationFrame(step);

      // Resize handler to refresh widths/heights on demand
      window.addEventListener('resize', () => {
        items.forEach(it => {
          const r = it.el.getBoundingClientRect();
          it.width = r.width; it.height = r.height;
        });
      });
    })();
  </script>
